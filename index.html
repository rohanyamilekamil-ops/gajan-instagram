// Auth State Observer - Isse refresh loop ruk jayega firebase.auth().onAuthStateChanged((user) => { const loader = document.getElementById('loader'); const authScreen = document.getElementById('authScreen'); const mainApp = document.getElementById('mainApp'); // Browser ko state update karne ka thoda time dein setTimeout(() => { if (user) { // Agar login ho gaya toh app dikhao authScreen.style.display = 'none'; mainApp.style.display = 'block'; if(loader) loader.style.display = 'none'; console.log("User active:", user.email); } else { // Agar login nahi hai toh login screen dikhao authScreen.style.display = 'flex'; mainApp.style.display = 'none'; if(loader) loader.style.display = 'none'; } }, 1000); // 1 second ka delay loop todne ke liye }); // POPUP Login Function function login() { const provider = new firebase.auth.GoogleAuthProvider(); firebase.auth().signInWithPopup(provider).catch((error) => { alert("Login Error: " + error.message); }); }
