// Purane signIn function ki jagah ye wala daalein function signIn() { const provider = new firebase.auth.GoogleAuthProvider(); // Popup ki jagah Redirect use karenge taaki refresh wala chakkar khatam ho jaye auth.signInWithRedirect(provider); } // Ye line zaroori hai redirect ke baad result pakadne ke liye auth.getRedirectResult().then((result) => { if (result.user) { console.log("Login Success:", result.user.displayName); } }).catch((error) => { console.error("Login Error:", error.message); });
