// Follow/Unfollow Functionality const followBtn = document.querySelector('button.bg-blue-500'); // Blue follow button const followerCountElement = document.querySelector('.flex.space-x-8 div:nth-child(2) span'); // Follower number let isFollowing = false; let followers = 0; if(followBtn) { followBtn.addEventListener('click', () => { if (!isFollowing) { // Follow logic isFollowing = true; followers++; followBtn.innerText = 'Following'; followBtn.classList.remove('bg-blue-500', 'text-white'); followBtn.classList.add('bg-gray-200', 'text-black'); } else { // Unfollow logic isFollowing = false; followers--; followBtn.innerText = 'Follow'; followBtn.classList.remove('bg-gray-200', 'text-black'); followBtn.classList.add('bg-blue-500', 'text-white'); } // Update the UI followerCountElement.innerText = followers; }); }
